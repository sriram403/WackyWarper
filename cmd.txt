# To connect cvat-to-training bucket to the instance
if it's new instance without gcsfuse
    1. curl -sLO https://github.com/GoogleCloudPlatform/gcsfuse/releases/download/v3.10.0/gcsfuse-3.10.0-1.x86_64.rpm
    2. sudo dnf install -y ./gcsfuse-3.10.0-1.x86_64.rpm

    3. gcsfuse --key-file=/home/ec2-user/cvat-access.json \
        --file-mode=777 --dir-mode=777 --implicit-dirs \
        cvat-to-training /home/ec2-user/Training_Model/Sriram/data

if it already has gcsfuse configured, then
    1. gcsfuse --file-mode=777 --dir-mode=777 --implicit-dirs cvat-to-training data
